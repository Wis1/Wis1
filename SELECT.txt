SELECT I.NAME, IE.SUMMARY
FROM ISSUESLISTS I
JOIN ISSUE IE ON I.ID= IE.ISSUESLIST_ID;

SELECT IE.ID, ISSUESLIST_ID, SUMMARY, DESCRIPTION, USER_ID_ASSIGNEDTO, U.FIRSTNAME, LASTNAME
FROM ISSUE IE
JOIN USERS U ON IE.USER_ID_ASSIGNEDTO=U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASKS_NUMBER
FROM USERS U JOIN ISSUE I on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO;
